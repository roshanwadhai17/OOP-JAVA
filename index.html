üöó Vehicle Hierarchy Example

// Parent Class (Superclass) class Vehicle { protected String brand; protected String model; protected int year; public Vehicle(String brand, String model, int year) { this.brand = brand; this.model = model; this.year = year; } public void start() { System.out.println(brand + " " + model + " is starting..."); } public void stop() { System.out.println(brand + " " + model + " has stopped."); } public void displayInfo() { System.out.println("Vehicle: " + brand + " " + model + " (" + year + ")"); } } // Child Class 1 class Car extends Vehicle { private int numberOfDoors; public Car(String brand, String model, int year, int numberOfDoors) { super(brand, model, year); // Call parent constructor this.numberOfDoors = numberOfDoors; } // Method specific to Car public void openTrunk() { System.out.println("Car trunk is opened."); } // Override parent method @Override public void start() { System.out.println("Car " + brand + " " + model + " engine started with key."); } } // Child Class 2 class Motorcycle extends Vehicle { private boolean hasSidecar; public Motorcycle(String brand, String model, int year, boolean hasSidecar) { super(brand, model, year); this.hasSidecar = hasSidecar; } // Method specific to Motorcycle public void wheelie() { System.out.println("Motorcycle is doing a wheelie!"); } @Override public void start() { System.out.println("Motorcycle " + brand + " " + model + " started with kick/button."); } } // Usage public class Main { public static void main(String[] args) { Car car = new Car("Toyota", "Camry", 2023, 4); Motorcycle bike = new Motorcycle("Harley", "Davidson", 2022, false); // Using inherited methods car.displayInfo(); // From Vehicle class car.start(); // Overridden in Car class car.openTrunk(); // Car-specific method bike.displayInfo(); // From Vehicle class bike.start(); // Overridden in Motorcycle class bike.wheelie(); // Motorcycle-specific method } }

üé≠ Polymorphism (Many Forms)
üé® What is Polymorphism?
The ability of a single interface to represent different underlying forms (data types). Same method name, different implementations.

üéµ Real-World Example: Musical Instruments

üéπ Piano
play() ‚Üí Keys pressed
üé∏ Guitar
play() ‚Üí Strings strummed
ü•Å Drums
play() ‚Üí Surface hit
üé∫ Trumpet
play() ‚Üí Air blown

Same method name "play()", but each instrument implements it differently!

üìã Types of Polymorphism

üïê Compile-Time Polymorphism
‚Ä¢ Method Overloading
‚Ä¢ Operator Overloading
‚Ä¢ Decided at compile time
‚è∞ Runtime Polymorphism
‚Ä¢ Method Overriding
‚Ä¢ Interface Implementation
‚Ä¢ Decided at runtime

üêæ Method Overloading Example

class Calculator { // Same method name, different parameters public int add(int a, int b) { return a + b; } public double add(double a, double b) { return a + b; } public int add(int a, int b, int c) { return a + b + c; } public String add(String a, String b) { return a + b; } } // Usage Calculator calc = new Calculator(); System.out.println(calc.add(5, 3)); // Calls int version ‚Üí 8 System.out.println(calc.add(5.5, 3.2)); // Calls double version ‚Üí 8.7 System.out.println(calc.add(1, 2, 3)); // Calls 3-parameter version ‚Üí 6 System.out.println(calc.add("Hello", "World")); // Calls String version ‚Üí HelloWorld

üé® Method Overriding Example

// Parent class class Shape { public void draw() { System.out.println("Drawing a generic shape"); } public double calculateArea() { return 0.0; } } // Child classes class Circle extends Shape { private double radius; public Circle(double radius) { this.radius = radius; } @Override public void draw() { System.out.println("Drawing a circle with radius " + radius); } @Override public double calculateArea() { return Math.PI * radius * radius; } } class Rectangle extends Shape { private double length, width; public Rectangle(double length, double width) { this.length = length; this.width = width; } @Override public void draw() { System.out.println("Drawing a rectangle " + length + "x" + width); } @Override public double calculateArea() { return length * width; } } // Demonstrating Runtime Polymorphism public class Main { public static void main(String[] args) { // Same reference type, different object types Shape[] shapes = { new Circle(5.0), new Rectangle(4.0, 6.0), new Circle(3.0) }; // Runtime polymorphism in action for (Shape shape : shapes) { shape.draw(); // Calls appropriate draw() method System.out.println("Area: " + shape.calculateArea()); System.out.println("---"); } } } /* Output: Drawing a circle with radius 5.0 Area: 78.53981633974483 --- Drawing a rectangle 4.0x6.0 Area: 24.0 --- Drawing a circle with radius 3.0 Area: 28.274333882308138 --- */

üèõÔ∏è Abstract Class Example

// Abstract class abstract class Animal { protected String name; protected int age; // Constructor in abstract class public Animal(String name, int age) { this.name = name; this.age = age; } // Concrete method (has implementation) public void sleep() { System.out.println(name + " is sleeping..."); } // Abstract method (no implementation) public abstract void makeSound(); public abstract void move(); // Concrete method public void displayInfo() { System.out.println("Name: " + name + ", Age: " + age); } } // Concrete class extending abstract class class Dog extends Animal { private String breed; public Dog(String name, int age, String breed) { super(name, age); this.breed = breed; } // Must implement abstract methods @Override public void makeSound() { System.out.println(name + " says: Woof! Woof!"); } @Override public void move() { System.out.println(name + " is running on four legs."); } public void wagTail() { System.out.println(name + " is wagging tail happily!"); } } class Cat extends Animal { public Cat(String name, int age) { super(name, age); } @Override public void makeSound() { System.out.println(name + " says: Meow! Meow!"); } @Override public void move() { System.out.println(name + " is walking gracefully."); } } // Usage public class Main { public static void main(String[] args) { // Animal animal = new Animal(); // Error! Cannot instantiate abstract class Dog dog = new Dog("Buddy", 3, "Golden Retriever"); Cat cat = new Cat("Whiskers", 2); dog.displayInfo(); // Inherited method dog.makeSound(); // Implemented abstract method dog.move(); // Implemented abstract method dog.wagTail(); // Dog-specific method dog.sleep(); // Inherited concrete method System.out.println("---"); cat.displayInfo(); cat.makeSound(); cat.move(); cat.sleep(); } }

üìã Interface Example

// Interface definition interface Drawable { // Abstract method (implicitly public and abstract) void draw(); void resize(double factor); // Default method (Java 8+) default void display() { System.out.println("Displaying the drawable object"); } // Static method (Java 8+) static void printInfo() { System.out.println("This is a drawable interface"); } } // Another interface interface Colorable { void setColor(String color); String getColor(); } // Class implementing multiple interfaces class Circle implements Drawable, Colorable { private double radius; private String color; public Circle(double radius) { this.radius = radius; this.color = "Black"; // default color } // Implement Drawable interface methods @Override public void draw() { System.out.println("Drawing a " + color + " circle with radius " + radius); } @Override public void resize(double factor) { radius *= factor; System.out.println("Circle resized. New radius: " + radius); } // Implement Colorable interface methods @Override public void setColor(String color) { this.color = color; } @Override public String getColor() { return color; } } class Rectangle implements Drawable, Colorable { private double length, width; private String color; public Rectangle(double length, double width) { this.length = length; this.width = width; this.color = "Black"; } @Override public void draw() { System.out.println("Drawing a " + color + " rectangle " + length + "x" + width); } @Override public void resize(double factor) { length *= factor; width *= factor; System.out.println("Rectangle resized to " + length + "x" + width); } @Override public void setColor(String color) { this.color = color; } @Override public String getColor() { return color; } } // Usage public class Main { public static void main(String[] args) { Circle circle = new Circle(5.0); Rectangle rectangle = new Rectangle(4.0, 6.0); // Using interface methods circle.setColor("Red"); circle.draw(); circle.resize(1.5); circle.display(); // Default method System.out.println("---"); rectangle.setColor("Blue"); rectangle.draw(); rectangle.resize(0.8); rectangle.display(); // Static method call Drawable.printInfo(); // Polymorphism with interfaces Drawable[] shapes = {new Circle(3.0), new Rectangle(2.0, 4.0)}; for (Drawable shape : shapes) { shape.draw(); } } }
üéØ Common Interview Questions

‚ùì Frequently Asked Questions

Q1: What is the difference between Abstract Class and Interface?
Feature Abstract Class Interface
Methods Can have abstract + concrete methods All abstract (before Java 8)
Variables Can have instance variables Only public static final
Inheritance Single inheritance (extends) Multiple inheritance (implements)
Constructor Can have constructors Cannot have constructors
Access Modifiers Any access modifier Methods are public by default
Q2: What is method overloading vs method overriding?

Method Overloading (Compile-time Polymorphism):
‚Ä¢ Same method name, different parameters
‚Ä¢ Happens in the same class
‚Ä¢ Resolved at compile time

Method Overriding (Runtime Polymorphism):
‚Ä¢ Same method signature in parent and child class
‚Ä¢ Child class provides specific implementation
‚Ä¢ Resolved at runtime
Q3: Can we override static methods?

Answer: No, we cannot override static methods. Static methods belong to the class, not to instances. However, we can have static methods with the same name in child classes (called method hiding).
Q4: What is the use of 'super' keyword?

‚Ä¢ super() - Call parent class constructor
‚Ä¢ super.method() - Call parent class method
‚Ä¢ super.variable - Access parent class variable
‚Ä¢ Must be the first statement in constructor (for super())
Q5: What is constructor chaining?

Constructor chaining is calling one constructor from another constructor using this() or super().
class Student { private String name; private int age; private String course; // Constructor 1 public Student() { this("Unknown", 0, "Not Enrolled"); // Calls constructor 3 } // Constructor 2 public Student(String name) { this(name, 0, "Not Enrolled"); // Calls constructor 3 } // Constructor 3 (Main constructor) public Student(String name, int age, String course) { this.name = name; this.age = age; this.course = course; } }

üéØ Complete Example Demonstrating All OOP Concepts
This example shows how all four OOP pillars work together in a practical scenario.
// Abstract class demonstrating Abstraction and Inheritance abstract class Employee { // Encapsulation - private fields private String name; private int id; protected double baseSalary; // Constructor public Employee(String name, int id, double baseSalary) { this.name = name; this.id = id; this.baseSalary = baseSalary; } // Concrete method public void displayInfo() { System.out.println("ID: " + id + ", Name: " + name + ", Base Salary: $" + baseSalary); } // Abstract method - must be implemented by subclasses public abstract double calculateSalary(); public abstract void work(); // Getters (Encapsulation) public String getName() { return name; } public int getId() { return id; } public double getBaseSalary() { return baseSalary; } } // Interface for bonus calculation interface BonusCalculator { double calculateBonus(); default void displayBonusInfo() { System.out.println("Bonus: $" + calculateBonus()); } } // Concrete class - Full Time Employee class FullTimeEmployee extends Employee implements BonusCalculator { private double monthlyAllowance; private int experienceYears; public FullTimeEmployee(String name, int id, double baseSalary, double monthlyAllowance, int experienceYears) { super(name, id, baseSalary); this.monthlyAllowance = monthlyAllowance; this.experienceYears = experienceYears; } // Method Overriding @Override public double calculateSalary() { return baseSalary + monthlyAllowance + calculateBonus(); } @Override public void work() { System.out.println(getName() + " is working full-time (40 hours/week)"); } @Override public double calculateBonus() { return baseSalary * 0.1 * experienceYears; // 10% per year of experience } } // Another concrete class - Part Time Employee class PartTimeEmployee extends Employee { private int hoursWorked; private double hourlyRate; public PartTimeEmployee(String name, int id, double baseSalary, int hoursWorked, double hourlyRate) { super(name, id, baseSalary); this.hoursWorked = hoursWorked; this.hourlyRate = hourlyRate; } @Override public double calculateSalary() { return hoursWorked * hourlyRate; } @Override public void work() { System.out.println(getName() + " is working part-time (" + hoursWorked + " hours)"); } } // Contractor class class Contractor extends Employee implements BonusCalculator { private int projectsCompleted; private double projectRate; public Contractor(String name, int id, double baseSalary, int projectsCompleted, double projectRate) { super(name, id, baseSalary); this.projectsCompleted = projectsCompleted; this.projectRate = projectRate; } @Override public double calculateSalary() { return projectsCompleted * projectRate + calculateBonus(); } @Override public void work() { System.out.println(getName() + " is working on contract projects"); } @Override public double calculateBonus() { return projectsCompleted > 5 ? 1000 : 0; // Bonus for completing >5 projects } } // Manager class demonstrating method overloading class EmployeeManager { // Method overloading - same name, different parameters public void promoteEmployee(Employee emp) { System.out.println("Promoting " + emp.getName() + " with standard promotion"); } public void promoteEmployee(Employee emp, double salaryIncrease) { System.out.println("Promoting " + emp.getName() + " with $" + salaryIncrease + " increase"); } public void promoteEmployee(Employee emp, String newPosition, double salaryIncrease) { System.out.println("Promoting " + emp.getName() + " to " + newPosition + " with $" + salaryIncrease + " increase"); } } // Main class demonstrating Polymorphism public class EmployeeManagementSystem { public static void main(String[] args) { // Creating different types of employees FullTimeEmployee fte = new FullTimeEmployee("Alice Johnson", 101, 5000, 1000, 3); PartTimeEmployee pte = new PartTimeEmployee("Bob Smith", 102, 0, 20, 25); Contractor contractor = new Contractor("Carol Brown", 103, 0, 6, 500); // Array of Employee references - Polymorphism Employee[] employees = {fte, pte, contractor}; System.out.println("=== EMPLOYEE MANAGEMENT SYSTEM ===\n"); // Polymorphism in action - same method calls, different behaviors for (Employee emp : employees) { emp.displayInfo(); // Inherited method emp.work(); // Overridden method (different for each type) System.out.println("Total Salary: $" + emp.calculateSalary()); // Check if employee is eligible for bonus if (emp instanceof BonusCalculator) { BonusCalculator bc = (BonusCalculator) emp; bc.displayBonusInfo(); } System.out.println("------------------------"); } // Demonstrating method overloading EmployeeManager manager = new EmployeeManager(); System.out.println("\n=== PROMOTIONS ==="); manager.promoteEmployee(fte); manager.promoteEmployee(pte, 500); manager.promoteEmployee(contractor, "Senior Contractor", 1000); } } /* Expected Output: === EMPLOYEE MANAGEMENT SYSTEM === ID: 101, Name: Alice Johnson, Base Salary: $5000.0 Alice Johnson is working full-time (40 hours/week) Total Salary: $7500.0 Bonus: $1500.0 ------------------------ ID: 102, Name: Bob Smith, Base Salary: $0.0 Bob Smith is working part-time (20 hours) Total Salary: $500.0 ------------------------ ID: 103, Name: Carol Brown, Base Salary: $0.0 Carol Brown is working on contract projects Total Salary: $4000.0 Bonus: $1000.0 ------------------------ === PROMOTIONS === Promoting Alice Johnson with standard promotion Promoting Bob Smith with $500.0 increase Promoting Carol Brown to Senior Contractor with $1000.0 increase */
